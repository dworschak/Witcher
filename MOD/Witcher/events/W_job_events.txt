# Hidden - Independant ruler - Give claim on county if have no claims and have weak neighbours
character_event = {
	id = job.1
	
	hide_window = yes
	
	min_age = 15
	capable_only = yes
	prisoner = no
	only_rulers = yes
	only_independent = yes
	ai = yes
	
	trigger = {
		job_chancellor = {
			NOT = { has_job_action = action_fabricate_claims }
		}
		any_realm_province = {
			any_neighbor_province = {
				NOT = { terrain = dryad_forest }
				owner = {
					NOT = { same_realm = ROOT }
					OR = {
						AND = {
							top_liege = { NOT = { relative_power = { who = ROOT power = 1 } } }
							independent = no
						}
						AND = {
							NOT = { relative_power = { who = ROOT power = 0.75 } }
							independent = yes
						}
					}
				}
			}
		}
		trait = ambitious
	}
	
	mean_time_to_happen = {
		months = 1000
		modifier = {
			factor = 2
			any_claim = {
				count = 1
				tier = count
			}
		}
		modifier = {
			factor = 1000
			any_claim = {
				count = 2
				tier = count
			}
		}
	}
	
	option = {
		name = OK
		random_realm_province = {
			limit = {
				any_neighbor_province = {
					NOT = { terrain = dryad_forest }
					owner = {
						NOT = { same_realm = ROOT }
						OR = {
							AND = {
								top_liege = { NOT = { relative_power = { who = ROOT power = 1 } } }
								independent = no
							}
							AND = {
								NOT = { relative_power = { who = ROOT power = 0.75 } }
								independent = yes
							}
						}
					}
				}
			}
			random_neighbor_province = {
				limit = {
					NOT = { terrain = dryad_forest }
					owner = {
						NOT = { same_realm = ROOT }
						OR = {
							AND = {
								top_liege = { NOT = { relative_power = { who = ROOT power = 1 } } }
								independent = no
							}
							AND = {
								NOT = { relative_power = { who = ROOT power = 0.75 } }
								independent = yes
							}
						}
					}
				}
				county = { add_claim = ROOT }
			}
		}
	}
}

# Chancellor promotes weak claim to strong
character_event = {
	id = job.10
	desc = EVTDESCjob.10
	picture = GFX_evt_spymaster
	
	only_rulers = yes
	min_age = 15
	capable_only = yes
	only_men = yes
	
	trigger = {
		job_chancellor = {
			diplomacy = 14
		}
		any_claim = {
				ROOT = { has_weak_claim = PREV }
		}
	}
	
	mean_time_to_happen = {
		months = 720
		modifier = {
			factor = 0.95
			diplomacy = 15
		}
		modifier = {
			factor = 0.95
			diplomacy = 16
		}
		modifier = {
			factor = 0.95
			diplomacy = 17
		}
		modifier = {
			factor = 0.95
			diplomacy = 18
		}
		modifier = {
			factor = 0.95
			diplomacy = 19
		}
		modifier = {
			factor = 0.95
			diplomacy = 20
		}
		modifier = {
			factor = 0.95
			diplomacy = 21
		}
		modifier = {
			factor = 0.95
			diplomacy = 22
		}
		modifier = {
			factor = 0.95
			diplomacy = 24
		}
		modifier = {
			factor = 0.95
			diplomacy = 26
		}
		modifier = {
			factor = 0.95
			diplomacy = 29
		}
	}
	
	option = {
		name = EVTOPTAjob.10
		random_claim = {
			limit = {
				ROOT = { has_weak_claim = PREV }
			}
			hidden_tooltip = { remove_claim = ROOT }
			add_pressed_claim = ROOT
		}
		scaled_wealth = -1.0
		if = { # Emperor or King
			limit = { primary_title = { higher_tier_than = DUKE } }
			prestige = -150
		}
		if = { # Duke
			limit = { primary_title = { tier = DUKE } }
			prestige = -100
		}
		if = { # Count or Patrician
			limit = {
				OR = {
					primary_title = { tier = COUNT }
					AND = {
						is_patrician = yes
						primary_title = { NOT = { higher_tier_than = COUNT } }
					}
				}
				primary_title = { tier = DUKE }
			}
			prestige = -50
		}
		ai_chance ={
			factor = 5
			modifier = { factor = 0 NOT = { prestige = 0 } }
			modifier = { factor = 0 NOT = { scaled_wealth = 1.25 } }
			modifier = { factor = 0 trait = content }
			modifier = { factor = 2 trait = ambitious }
		}
	}
	option = {
		name = EVTOPTBjob.10
		piety = 10
		ai_chance ={
			factor = 1
		}
	}
}
