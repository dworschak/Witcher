			FROM = {

	}

			limit = {
				ai = no
				OR = {
					has_landed_title = e_il-khanate

					has_landed_title = e_mongol_empire
					liege = { has_landed_title = e_il-khanate }
					liege = { has_landed_title = e_golden_horde }

				}
			}

		}
	}

		name = "EVTOPTA60100"
