	OR = { 
		check_variable = { which = rank1 which = FROM }
		
		AND = {
			is_variable_equal = { which = rank1 which = FROM }
			check_variable = { which = rank2 which = FROM }
		}
		
		AND = {
			is_variable_equal = { which = rank1 which = FROM }
			is_variable_equal = { which = rank2 which = FROM }
			check_variable = { which = kick_rank which = FROM }
		}
	}
}
