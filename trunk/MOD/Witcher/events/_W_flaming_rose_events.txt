# The Grand Master humbly requests the right to build a castle in your demesne
character_event = {
	id = flaming_rose.40
	desc = "Being Grandmaster ain't fun"
	picture = GFX_evt_crusaders
	border = GFX_event_normal_frame_religion
	
	hide_window = yes
	
	only_playable = yes
	
	trigger = {
		religion = eternal_fire
		
		is_heretic = no
		
		d_flaming_rose = {
			holder_scope = {
				wealth = 300
				NOT = {
					realm_size = 10
				}
				NOT = {
					num_of_count_titles = 1
				}
			}
		}
		
		any_demesne_province = {
			has_empty_holding = yes
			empire = {
				title = e_the_north
			}
		}
		
		has_dlc = "Sons of Abraham"
	}
	
	mean_time_to_happen = {
		months = 240
		modifier = {
			factor = 0.5
			d_flaming_rose = {
				holder_scope = {
					NOT = { demesne_size = 1 }
				}
			}
		}
		modifier = {
			factor = 0.5
			d_flaming_rose = {
				holder_scope = {
					NOT = { demesne_size = 2 }
				}
			}
		}
		modifier = {
			factor = 2.0
			d_flaming_rose = {
				holder_scope = {
					realm_size = 5
				}
			}
		}
	}
	
	immediate = {
		random_demesne_province = {
			limit = {
				has_empty_holding = yes
				empire = {
					title = e_the_north
				}
			}
			province_event = {
				id = flaming_rose.41
			}
		}
	}
	
	option = {
		name = "OK"
	}
}
